\setlist[chapteroutlineenv,1]{label=\arabic*} % Level 1: 1, 2, 3, ...
\setlist[chapteroutlineenv,2]{label*=.\arabic*} % Level 2: 1.1, 1.2, ...




\newenvironment{chapteroutline}{
    \renewcommand{\outlinei}{chapteroutlineenv}
    \renewcommand{\outlineii}{chapteroutlineenv}
    \begin{outline}
}{
    \end{outline}
    \renewcommand{\outlinei}{itemize}
    \renewcommand{\outlineii}{itemize}
}
